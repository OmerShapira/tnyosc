00130 int gettimeofday(struct timeval *tv, struct timezone *tz)
00131 {
00132   FILETIME ft;
00133   int64_t tmpres = 0;
00134   TIME_ZONE_INFORMATION tz_winapi;
00135   int rez = 0;
00136 
00137   if (!tv && !tz) {
00138     return -1;
00139   }
00140 
00141   ZeroMemory(&ft, sizeof(ft));
00142   ZeroMemory(&tz_winapi, sizeof(tz_winapi));
00143 
00144   GetSystemTimeAsFileTime(&ft);
00145   tmpres = ft.dwHighDateTime;
00146   tmpres <<= 32;
00147   tmpres |= ft.dwLowDateTime;
00148   // converting file time to unix epoch
00149   tmpres /= 10;  // convert into microseconds
00150   tmpres -= DELTA_EPOCH_IN_MICROSECS; 
00151 
00152   if (tv) {
00153     tv->tv_sec = (time_t)(tmpres * 0.000001);
00154     tv->tv_usec =(tmpres % 1000000);
00155   }
00156 
00157   // _tzset(), don't work properly, so we use GetTimeZoneInformation
00158   if (tz) {
00159     rez = GetTimeZoneInformation(&tz_winapi);
00160     tz->tz_dsttime = rez;
00161     tz->tz_minuteswest = tz_winapi.Bias + 
00162       (rez == 2 ? tz_winapi.DaylightBias : 0);
00163   }
00164 
00165   return 0;
00166 }

00172 uint64_t get_current_ntp_time()
00173 {
00174   // time between 1-1-1900 and 1-1-1950
00175   static const uint64_t epoch = 2208988800UL;
00176   // max value of NTP fractional part
00177   static const uint64_t ntp_scale = 4294967295UL;
00178 
00179   struct timeval tv;
00180   gettimeofday(&tv, NULL);
00181   uint64_t tv_ntp = tv.tv_sec + epoch;
00182   // convert tv_usec to a fraction of a second
00183   uint64_t tv_usecs = (tv.tv_usec * ntp_scale) / 1000000UL;
00184   return ((tv_ntp << 32) | tv_usecs);
00185 }
